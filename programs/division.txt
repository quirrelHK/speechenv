import java.util.Scanner;
class demopattern{
  public static void main(String[] args){
    Scanner s = new Scanner(System.in);
    System.out.println("Enter size: ");
    int size = s.nextInt();

    System.out.println("-------Pattern-------");
    for(int i = 0; i < size; ++i){
      for(int j = 0; j <= i; ++j){
        System.out.print(i + 1);
      }
      System.out.println();
    }
  }
}
